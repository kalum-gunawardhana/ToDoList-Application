CREATE TABLE active_tasks (
    task_id INT AUTO_INCREMENT PRIMARY KEY,
    task_title VARCHAR(100) NOT NULL,
    task_description TEXT,
    created_date DATE NOT NULL
);

CREATE TABLE completed_tasks (
    task_id INT AUTO_INCREMENT PRIMARY KEY,
    task_title VARCHAR(100) NOT NULL,
    task_description TEXT,
    completion_date DATE NOT NULL
);
